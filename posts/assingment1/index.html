#include <stdio.h>  
#include <time.h>   
void multiplyMatrices(int first[][2], int second[][2], int result[][2], int r1, int c1, int r2, int c2) {
    for (int i = 0; i < r1; i++) {  // Loop over each row of the first matrix
        for (int j = 0; j < c2; j++) {  // Loop over each column of the second matrix
            result[i][j] = 0;  // Initialize current element of result matrix to 0
            for (int k = 0; k < c1; k++) {  // Loop over columns of the first matrix (rows of the second matrix)
                result[i][j] += first[i][k] * second[k][j];  
            }
        }
    }
}

void displayMatrix(int matrix[][2], int row, int column) {
    for (int i = 0; i < row; i++) {  // Loop through each row of the matrix
        for (int j = 0; j < column; j++) {  // Loop through each column of the matrix
            printf("%d  ", matrix[i][j]);  // Print current element followed by a space
        }
        printf("\n");  // Print a newline after each row is printed
    }
}

int main() {
    int firstMatrix[2][2] = {{1, 2}, {3, 4}};  // Define and initialize first 2x2 matrix
    int secondMatrix[2][2] = {{2, 0}, {1, 2}};  // Define and initialize second 2x2 matrix
    int resultMatrix[2][2];  // Define result matrix

    // Start timing
    clock_t start = clock();  // Record start time
    
    multiplyMatrices(firstMatrix, secondMatrix, resultMatrix, 2, 2, 2, 2);  
    
    // End timing
    clock_t end = clock();  // Record end time

    // Calculate and print execution time
    double time_spent = (double)(end - start) / CLOCKS_PER_SEC;  
    printf("Execution Time: %f seconds\n", time_spent);  // Output execution time

    // Display result matrix
    displayMatrix(resultMatrix, 2, 2);  // Call display matrix function

    return 0;  // End of program
}

def multiply_matrices(a, b):
    # Initialize the result matrix with zeros
    result = [[0 for _ in range(len(b[0]))] for _ in range(len(a))]

    # Perform matrix multiplication
    for i in range(len(a)):
        for j in range(len(b[0])):
            for k in range(len(b)):
                result[i][j] += a[i][k] * b[k][j]

    return result

# Example matrices
a = [[1, 2], [3, 4]]
b = [[2, 0], [1, 2]]

# Calculate the product of matrices a and b
result = multiply_matrices(a, b)

# Print the result matrix
print("Result of matrix multiplication:")
for row in result:
    print(row)

Analysis

Compiled vs. Interpreted Languages: C is a compiled language, meaning it is converted into machine code prior to execution, which greatly enhances its efficiency. In contrast, Python is an interpreted language, typically processed line by line during runtime, which can result in reduced execution speed.

Memory Management: C offers more granular control over memory management, enabling programmers to manually manage memory allocation and deallocation, thereby optimizing performance. Python, however, employs automatic memory management, which simplifies coding but may compromise performance under certain conditions.

Optimization Levels: The C compiler, such as GCC, supports various optimization options that can significantly improve performance. Python’s performance optimizations, on the other hand, often depend on third-party libraries and tools, which might not offer the same level of control as C.
